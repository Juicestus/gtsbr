No Signal

Uses JS to sent a div around another div

Code


<script>
x=4;
y=4;
s=1;
xc = s;
yc = s;
c=0;
colors = ["red","green","blue","yellow"]
setInterval(function(){
        if ((648 <= x && x <= 652) || (-2 <= x && x <= 2)) {
                xc = -1/xc;
                c++;
        }

        if ((398 <= y && y <= 402) || (-2 <= y && y <= 2)) {
                yc = -1/yc;
                c++;
        }
        x+=xc;
        y+=yc;
        color = colors[c % colors.length]
        set = `<div class="box" style="background-color: ${color}; margin-left:${x}px; margin-top:${y}px;"><h2>No Signal</h2></div>`
        document.getElementById("window").innerHTML = set;

}, 1);
</script>
